-- Pryecto fullStack Django Vue + MySQL

CONFIGURACIÓN DEL ENTORNO
    python -m venv envFullStack

INSTALACION DE DJANGO + RESTFRAMEWORK
    pip install djangorestframework
    -- nos intalará últimas versiones
INSTALACION DE CORS HEADERS
    pip install django-cors-headers

CREAR PROYECTO DE DJANGO
    django-admin startproject DjangoAPI .

LEVANTAR EL SERVIDOR
    python manage.py runserver

CREAR APP
    django-admin startapp EmployeeApp

ASIGNAR APPS A settings.py INTALLED_APPS
    'rest_framework',
    'corsheaders',
    'EmployeeApp.apps.EmployeeappConfig'

ASIGNAR MIDELWARE CORSHEADERS:
    'corsheaders.middleware.CorsMiddleware',

AGREGAR LA LINEA CORS ORIGIN ALLOW ALL entre INSTALLED_APPS Y MIDDLEWARE
    CORS_ORIGIN_ALLOW_ALL = True


CREACIÓN DE MODELOS:
    Departaments
    Employees

CREACIÓN DE BASE DE DATOS DE MySQL 
    nombre de la conexión: mytestmysql
    usuario: testadmin@mytestmysql
    clave: Prueba1234
    (Realmente uso root/PasswordPrueba1...)
    nombre de la base: mytestdb

INSTALACIÓN DEL DRIVER DE MySQL
    pip install pymysql

ASIGNAR LA CONEXIÓN EN settings.py

--Es necesario para mysql agregar las siguientes 2 líneas
-- antes de la línea DATABASES = {}

    import pymysql
    pymysql.install_as_MySQLdb()
DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.mysql',
        'NAME': 'mytestdb',

        'PORT': '3307'
    }
}

REALIZAR MIGRACIONES PARA MODELO EmployeeApp

    python manage.py mekemigrations EmployeeApp

REALIZAR MIGRATE DEL MODELO EmployeeApp
    python manage.py migrate EmployeeApp
